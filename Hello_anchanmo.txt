def GetMinMax(data):

    min_value = max_value = data[0]

    for num in data:
        if num < min_value:
            min_value = num
        elif num > max_value:
            max_value = num

    return min_value, max_value

min_value, max_value = GetMinMax([5, 6, 3, 9, 8, 1, 4])
print(f최소값: {min_value})
print(f최대값: {max_value})
